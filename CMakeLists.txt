cmake_minimum_required(VERSION 2.8.9)
project(PothosBlocks)
enable_language(CXX)

if (NOT POTHOS_MODULE_PATH)

include_directories(${CMAKE_CURRENT_SOURCE_DIR}/network/udt4/src)
